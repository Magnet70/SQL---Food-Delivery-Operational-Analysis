        "import sqlite3\n",
        "import pandas as pd\n",
        "import random\n",
        "from datetime import datetime, timedelta\n",
        "\n",
        "# Set random seed for reproducibility\n",
        "random.seed(42)\n",
        "\n",
        "# Create in-memory SQLite database\n",
        "conn = sqlite3.connect(\":memory:\")\n",
        "cursor = conn.cursor()\n",
        "\n",
        "# Helper: generate random datetime between years\n",
        "def random_date(start_year=2020, end_year=2024):\n",
        "    start = datetime(start_year, 1, 1)\n",
        "    end = datetime(end_year, 12, 31)\n",
        "    delta = end - start\n",
        "    return (start + timedelta(days=random.randint(0, delta.days))).strftime(\"%Y-%m-%d\")"
      ]
    },

        "# Create tables for customers, restaurants, couriers, and orders\n",
        "cursor.executescript(\"\"\"\n",
        "                        CREATE TABLE customers (\n",
        "                        customer_id INTEGER PRIMARY KEY,\n",
        "                        name TEXT,\n",
        "                        city TEXT,\n",
        "                        registration_date DATE\n",
        "                        );\n",
        "\n",
        "                        CREATE TABLE restaurants (\n",
        "                        restaurant_id INTEGER PRIMARY KEY,\n",
        "                        name TEXT,\n",
        "                        city TEXT,\n",
        "                        cuisine TEXT\n",
        "                        );\n",
        "\n",
        "                        CREATE TABLE couriers (\n",
        "                        courier_id INTEGER PRIMARY KEY,\n",
        "                        name TEXT,\n",
        "                        city TEXT,\n",
        "                        rating REAL\n",
        "                        );\n",
        "\n",
        "                        CREATE TABLE orders (\n",
        "                        order_id INTEGER PRIMARY KEY,\n",
        "                        customer_id INTEGER,\n",
        "                        restaurant_id INTEGER,\n",
        "                        courier_id INTEGER,\n",
        "                        order_date DATE,\n",
        "                        delivery_time INTEGER, -- in minutes\n",
        "                        order_amount REAL,\n",
        "                        customer_rating INTEGER,\n",
        "                        FOREIGN KEY (customer_id) REFERENCES customers(customer_id),\n",
        "                        FOREIGN KEY (restaurant_id) REFERENCES restaurants(restaurant_id),\n",
        "                        FOREIGN KEY (courier_id) REFERENCES couriers(courier_id)\n",
        "                        );\n",
